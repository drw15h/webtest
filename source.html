@echo off
title Sort
setlocal EnableDelayedExpansion

set sortlist=
set totalist=0
cls
echo Add positive number (D if done)
echo.
echo.
:list
set newlist=
set/p newlist=Number:
if not defined newlist goto list
if !newlist! == D goto sort
set sortlist=!sortlist! !newlist!
set/a totalist=!totalist!+1
goto list

cls
echo Sorting...
set perfect=
:sort
set sortedlist=
set high=0
for %%a in (!sortlist!) do (
if %%a GTR !high! set high=%%a
)
for %%a in (!sortlist!) do (
if not %%a == !high! (set sortedlist=%%a !sortedlist!) else set perfect=!perfect! %%a
)
set sortlist=!sortedlist!
if not !totalist! == 0 (set/a totalist=!totalist!-1) & goto sort

cls
for %%a in (!perfect!) do (
echo %%a
)
echo.
echo Sorted successfully...
echo.
pause
exit
